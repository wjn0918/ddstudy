打包指定源码
```
     <plugins>
            <plugin>
                <groupId>org.apache.maven.plugins</groupId>
                <artifactId>maven-compiler-plugin</artifactId>
                <configuration>
                    <source>8</source>
                    <target>8</target>
                    <includes>
                        <include>
                            HelloWorld.java
                        </include>
                    </includes>
                </configuration>
            </plugin>
        </plugins>
```
